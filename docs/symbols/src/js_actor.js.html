1 /**
  2  * @class An actor
  3  * @param configs
  4  * @constructor
  5  */
  6 ActorJS.Actor = function (configs) {
  7 
  8     configs = configs || {};
  9 
 10     this.id = configs.id || "";
 11 
 12     /* Path to this actor within the hierarchy
 13      */
 14     this.path = configs.parent && configs.parent.id != "" ? (configs.parent.path + configs.ctx.pathSeparator + this.id) : this.id;
 15 
 16     /* Parent actor
 17      */
 18     this._parent = configs.parent;
 19 
 20     /*
 21      */
 22     this._loading = false;
 23 
 24     /* Buffer incoming calls when set
 25      */
 26     this._locked = false;
 27 
 28     /*
 29      */
 30     this._loaded = !!configs.loaded;
 31 
 32     /* Calls buffered while this actor is loading
 33      */
 34     this._callBuffer = [];
 35 
 36     /* Subscriptions buffered while this actor is loading
 37      */
 38     this._subsBuffer = [];
 39 
 40     /* Context shared among all actors
 41      */
 42     this._ctx = configs.ctx;
 43 
 44     /* Child actors
 45      */
 46     this._actors = {};
 47 
 48     /* Internally managed actor IDs
 49      */
 50     this._defaultActorIds = new ActorJS._Map("__");
 51 
 52     /* Holds the most recent publication for each topic
 53      */
 54     this._publications = {};
 55 
 56     /* Map of subscriptions for each topic.
 57      */
 58     this._topicSubs = {};
 59 
 60     /* objects - objects of any type, put on the actor for use by child actors
 61      */
 62     this._objects = {};
 63 
 64     this.inNewThread = configs.inNewThread || configs.newThread;
 65 };
 66 
 67 /* Pool of reusable IDs for subscription handles
 68  */
 69 ActorJS._subHandles = new ActorJS._Map();
 70 
 71 /**
 72  * Grab lock, causing buffering of incoming calls while held
 73  */
 74 ActorJS.Actor.prototype.lock = function () {
 75     this._locked = true;
 76 };
 77 
 78 /**
 79  * Release lock, and if actor is loaded, execute buffered calls
 80  */
 81 ActorJS.Actor.prototype.unlock = function () {
 82     if (this._locked) {
 83         this._locked = false;
 84         if (this._loaded) { // Otherwise calls will un-buffer when load completes
 85             this._doBufferedCalls();
 86             this._doBufferedSubscriptions();
 87         }
 88     }
 89 };

 91 /**
 92  * Makes asynchronous call to a method on either the actor or a target actor
 93  *
 94  *
 95  * <p>Example - calling the actor's {@link #addActor} method:</p>
 96  *
 97  * <pre>
 98  * #call("addActor",{
 99                 *      type: "myType",
100                 *      id: "myActorInstance",
101                 *      someActorConfig: "foo",
102                 *      otherActorConfig: "bar"
103                 * });
104  * </pre>
105  *
106  * <p>Example - calling the actor's {@link #removeActor} method:</p>
107  *
108  * <pre>
109  * #call("removeActor", {
110                 *      id: "myActorInstance"
111                  * });
112  * </pre>
113  *
114  * <p>Example - calling a method on an actor:</p>
115  *
116  * <pre>
117  * #call("myActorInstance.someMethod", {
118                 *              someMethodParam: "foo",
119                 *              otherMethodParam: "bar"
120                 * });
121  * </pre>
122  *
123  * @param {String} method Name of method on either the actor or one of its actors
124  * @param {JSON} params Params for target method
125  * @return {actor} This actor
126  */
127 ActorJS.Actor.prototype.call = function (method, params) {
128 
129     params = params || {};
130 
131     /*-----------------------------------------------------------------------------------
132      * If actor locked or not yet loaded, then buffer the call until the actor is ready
133      *---------------------------------------------------------------------------------*/
134 
135     if ((!this._loaded && !this._loading) || this._locked) {
136 
137         this._callBuffer.unshift({
138             method:method,
139             params:params
140         });
141 
142         return this;
143     }
144 
145     var slashIdx = method.indexOf(this._ctx.pathSeparator);
146 
147     if (slashIdx > 0) { // Calling child actor
148 
149         var actorId = method.substr(0, slashIdx);
150 
151         var actor = this._actors[actorId];
152 
153         /*----------------------------------------------------------------------------
154          * Call method on actor
155          *--------------------------------------------------------------------------*/
156 
157         if (actor) {
158 
159             /* Actor in Worker
160              */
161             if (actor.worker) {
162 
163                 actor.worker.postMessage({
164                     action:"call",
165                     method:method,
166                     params:params
167                 });
168 
169                 return this;
170             }
171 
172             var subPath = method.substr(slashIdx + 1);
173             var slashIdx2 = subPath.indexOf(this._ctx.pathSeparator);
174             var actorMethod = (slashIdx2 > 0) ? subPath.substr(slashIdx2 + 1) : subPath;
175 
176             actor.call(subPath, params);
177 
178         } else {
179 
180             if (this._parent) {
181 
182                 /* Actor not found - calling a parent actor
183                  */
184                 this._parent.call(method, params);
185 
186             } else {
187 
188                 throw "method not found: " + method;
189             }
190         }
191 
192     } else {
193 
194 
195         /* No slash in method name - calling a method on this actor
196          */
197 
198         var fn = this[method];
199 
200         if (!fn) {
201 
202             if (this._parent) {
203 
204                 /* Actor not found - calling a parent actor
205                  */
206                 this._parent.call(method, params);
207 
208                 return this;
209 
210             } else {
211 
212                 throw "method not found: " + method;
213             }
214 
215             //throw "method not found on actor '" + this.id + "': " + method;
216 
217         } else {
218 
219             fn.call(this, params);
220         }
221     }
222 
223     return this;
224 };

227 /**
228  * Adds an actor to this actor
229  *
230  * @param params
231  * @param {String} params.type Actor type
232  * @param {String} params.include Actor JSON include, which maps to a JSON fragment
233  * @param {String} params.id ID for actor, unique among actors on this actor - falls back on the value of params.type when omitted
234  * @param {String} params.existsOK When true and actor already exists, the call will be ignored without throwing an exception
235  * @return {*} Proxy actor through which method calls may be made, deferred until the actor exists
236  * @throws exception If actor already exists and existsOK not true
237  */
238 ActorJS.Actor.prototype.addActor = function (params) {
239 
240     if (!params) {
241         throw "argument expected: params";
242     }
243 
244     var actorId = params.id;
245 
246     if (!actorId) {
247         actorId = this._defaultActorIds.addItem({}); // Automatic actor ID
248 
249     } else {
250 
251         if (this._actors[actorId]) {
252             if (params.existsOK) {
253                 return actorId;
254             }
255 
256             throw "actor already exists: " + actorId;
257         }
258     }
259 
260     if (!params.type && !params.include) {
261         throw "param expected: type or include";
262     }
263 
264     var taskId = actorId + ".create";
265 
266     this.publish("task.started", {
267         taskId:taskId,
268         description:"Creating actor '" + actorId + "'"
269     });
270 
271     var actor = new ActorJS.Actor({
272         id:actorId,
273         parent:this,
274         ctx:this._ctx,
275         inNewThread:this.inNewThread
276     });
277 
278     this._actors[actorId] = actor;
279 
280     var self = this;
281 
282     /* Include JSON
283      */
284     if (params.include) {
285         this._ctx.loadInclude(params.include, // Load JSON
286             function (json) {
287                 var type = json.type; // Root of include cannot itself be an include - can only be a type
288                 if (!type) {
289                     throw "param expected: type";
290                 }
291                 self._ctx.loadType(type,
292                     function (clazz) {
293                         create(clazz, json);
294                     },
295                     error); // Load actor class for JSON root node
296             },
297             function (err) {
298                 error("failed to include actor " + params.include + ": " + err);
299             });
300         return;
301 
302     } else {
303 
304         /* Inline
305          */
306         this._ctx.loadType(params.type,
307             function (clazz) {
308                 create(clazz, params);
309             },
310             function (err) {
311                 error("failed to add actor type " + params.type + ": " + err);
312             }); // Load actor class
313     }
314 
315     function create(clazz, params) {
316         var cfg = {
317             id:actorId
318         };
319         for (var key in params) {
320             if (params.hasOwnProperty(key)) {
321                 cfg[key] = params[key]
322             }
323         }
324         try {
325             /* Call type constructor, which is more like an initialisation
326              * method than a constructor, where it augments the new actor
327              * instance with type-specific methods etc.
328              */
329             actor._loading = true; // Don't buffer calls on actor made by its constructor
330             clazz.call(actor, cfg);
331 
332             /**
333              * Add child actors
334              */
335             if (params.actors) {
336                 var actors = params.actors;
337                 for (var i = 0, len = actors.length; i < len; i++) {
338                     actor.call("addActor", actors[i]);
339                 }
340             }
341 
342             actor._loading = false;
343 
344         } catch (err) {
345             self.publish("task.failed", {
346                 taskId:taskId,
347                 error:err
348             });
349             self.publish("error", {
350                 error:err
351             });
352             return;
353         }
354         actor._loaded = true;
355         if (!actor._locked) {
356             actor._doBufferedCalls();
357             actor._doBufferedSubscriptions();
358         }
359         self.publish("task.finished", {
360             taskId:taskId
361         });
362     }
363 
364     function error(err) {
365         self.publish("task.failed", {
366             taskId:taskId,
367             error:err
368         });
369         self.publish("error", {
370             error:err
371         });
372     }
373 
374 
375     return actorId;
376 };
377 
378 ActorJS.Actor.prototype._doBufferedCalls = function () {
379     var call;
380     while (this._callBuffer.length > 0) {
381         call = this._callBuffer.pop();
382         this.call(call.method, call.params);
383     }
384 };
385 
386 
387 ActorJS.Actor.prototype._doBufferedSubscriptions = function () {
388     var sub;
389     while (this._subsBuffer.length > 0) {
390         sub = this._subsBuffer.pop();
391         this._subscribe(sub.topic, sub.handler, sub.handle);
392     }
393 };

396 /**
397  * Deletes an actor. Silently ignores if actor does not exist.
398  *
399  * @param {String} id ID of target actor
400  */
401 ActorJS.Actor.prototype.removeActor = function (params) {
402 
403     var actorId = params.id;
404 
405     if (!actorId) {
406         throw "param expected: id";
407     }
408 
409     var actor = self._actors[actorId];
410 
411     if (!actor) {
412         return;
413     }
414 
415     if (actor._destroy) {
416         actor._destroy();
417     }
418 
419     delete this._actors[actorId];
420 
421     if (this._defaultActorIds.items[actorId]) {
422         this._defaultActorIds.removeItem(actorId);
423     }
424 };

427 /**
428  * On to a topic on this actor and returns a handle to the subscription. Before the handle is
429  * returned, the subscription is immediately notified of the most recent publication on the topic,
430  * if one exists.
431  *
432  * @param {String} topic Topic name - "*" to on to all topics on this actor
433  * @param {Function} handler Topic handler
434  * @return {String} Subscription handle
435  */
436 ActorJS.Actor.prototype.on = function (topic, handler) {
437 
438     var handle = ActorJS.Actor._subHandles.addItem({
439         topic:topic
440     });
441 
442     for (var actor = this; actor; actor = actor._parent) {
443         actor._subscribe(topic, handler, handle);
444     }
445 
446     return handle;
447 };
448 
449 ActorJS.Actor.prototype._subscribe = function (topic, handler, handle) { // TODO: handle workers
450 
451     var slashIdx = topic.indexOf(this._ctx.pathSeparator);
452 
453     if (slashIdx > 0) {
454 
455         var actorId = topic.substr(0, slashIdx);
456 
457         var actor = this._actors[actorId];
458 
459         /* Buffer subscription if actor not yet existing
460          */
461         if (!actor) {
462 
463             this._subsBuffer.unshift({
464                 topic:topic,
465                 handler:handler,
466                 handle:handle
467             });
468 
469         } else {
470 
471             actor._subscribe(topic.substr(slashIdx + 1), handler, handle);
472         }
473 
474     } else {
475 
476         var subs = this._topicSubs[topic];
477 
478         if (!subs) {
479 
480             subs = {           // Subscriptions for this event topic
481                 handlers:{}, // Handler function for each subscriber
482                 numSubs:0                      // Count of subscribers for the event topic
483             };
484 
485             this._topicSubs[topic] = subs;
486         }
487 
488         subs.handlers[handle] = handler;
489 
490         subs.numSubs++;                     // Bump count of subscribers to the event
491 
492         var publication = this._publications[topic];
493 
494         if (publication) {
495             handler(publication);
496         }
497     }
498 
499     return handle;
500 };

503 /**
504  * Unsubscribe to an event on this actor
505  * @param {String} handle Subscription handle
506  */
507 ActorJS.Actor.prototype.unsubscribe = function (handle) {
508 
509     var sub = ActorJS.Actor._subHandles.items[handle];
510 
511     if (sub) {
512 
513         ActorJS.Actor._subHandles.removeItem(handle);
514 
515         var topic = sub.topic;
516 
517         this._unsubscribe(topic, handle);
518     }
519 
520 
521     for (var actor = this; actor; actor = actor._parent) {
522         actor._unsubscribe(topic, handle);
523     }
524 };

527 ActorJS.Actor.prototype._unsubscribe = function (topic, handle) {
528 
529     var slashIdx = topic.indexOf(this._ctx.pathSeparator);
530 
531     if (slashIdx > 0) {
532 
533         var actorId = topic.substr(0, slashIdx);
534 
535         var actor = this._actors[actorId];
536 
537         if (actor) {
538 
539             actor._unsubscribe(topic.substr(slashIdx + 1), handle);
540         }
541 
542     } else {
543 
544         var subs = this._topicSubs[topic];
545 
546         if (subs) {
547 
548             delete subs.handlers[handle];
549 
550             subs.numSubs--;
551             subs.numSubs--;
552         }
553     }
554 };

557 /**
558  * Publishes a topic on the actor
559  *
560  * @param {String} method Name of topic
561  * @param {Object} params Optional params for publication.
562  */
563 ActorJS.Actor.prototype.publish = function (topic, params) {
564 
565     params = params || {};
566 
567     this._publications[topic] = params;
568 
569     var subs = this._topicSubs[topic];
570 
571     if (subs) {
572 
573         if (subs.numSubs > 0) {             // Don't handle if no subscribers
574 
575             var handlers = subs.handlers;
576 
577             for (var handle in handlers) {
578                 if (handlers.hasOwnProperty(handle)) {
579 
580                     handlers[handle](params, topic);
581                 }
582             }
583         }
584     }
585 
586     /* Notify wildcard subscriptions
587      */
588 
589     subs = this._topicSubs["*"];
590 
591     if (subs) {
592 
593         if (subs.numSubs > 0) {             // Don't handle if no subscribers
594 
595             var handlers = subs.handlers;
596 
597             for (var handle in handlers) {
598                 if (handlers.hasOwnProperty(handle)) {
599 
600                     handlers[handle](params, topic);
601                 }
602             }
603         }
604     }
605 };

607 /**
608  * Adds a object for use by child actors
609  * @param {String} objectId
610  * @param {Object} object
611  */
612 ActorJS.Actor.prototype.setResource = function (objectId, object) {
613 
614     if (this._objects[objectId]) {
615         throw "object already exists: " + objectId;
616     }
617 
618     if (object) {
619         this._objects[objectId] = object;
620 
621     } else {
622         delete this._objects[objectId];
623     }
624 };

627 /**
628  * Gets a resource provided via the parent actor
629  * @param {String} objectId
630  */
631 ActorJS.Actor.prototype.getResource = function (objectId) {
632 
633     var object;
634 
635     for (var actor = this._parent; actor && !object; actor = actor._parent) {
636         object = actor._objects[objectId];
637     }
638 
639     return object;
640 };

643 /**
644  * Deletes actors, subscriptions and child actors
645  */
646 ActorJS.Actor.prototype.clear = function () {
647 
648     this._types = {};
649 
650     for (var actorId in this._actors) {
651         if (this._actors.hasOwnProperty(actorId)) {
652             this.removeActor({ id:actorId }); // Calls actor destructors
653         }
654     }
655 
656     this._topicSubs = {};
657 
658     ActorJS.Actor._subHandles.clear();
659 
660     for (var actorId in this._actors) {
661         if (this._actors.hasOwnProperty(actorId)) {
662             this.removeActor({ id:actorId });
663         }
664     }
665 };
